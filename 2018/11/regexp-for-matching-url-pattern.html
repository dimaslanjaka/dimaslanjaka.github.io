Regexp Pattern Untuk mencocokkan semua jenis URL, kode berikut seharusnya berfungsi: 


<?php
    $regex = “((https?|ftp)😕/)?”; // SCHEME
    regex .= "([a-z0-9+!*(),;?&amp;=.-]+(:[a-z0-9+!*(),;?&=$.-]+)?@)?"; // User and Pass
    $regex .= “([a-z0-9-.]).(([a-z]{2,4})|([0-9]{1,3}.([0-9]{1,3}).([0-9]{1,3})))"; // Host or IP
    $regex .= “(:[0-9]{2,5})?”; // Port
    regex.="(/([a−z0−9+regex .= "(/([a-z0-9+regex.="(/([a−z0−9+_%-].?)+)/?”; // Path
    regex .= "(\?[a-z+&amp;\$_.-][a-z0-9;:@&amp;%=+/.-]*)?"; // GET Query
    regex .= "(#[a-z_.-][a-z0-9+%.-]*)?"; // Anchor
?>
