PHP Usage Example
PHP Normal Usage
$websocket = new WMI_websocket(false, false);
/* Data Example */
$websocket->createData([
  'pendapatan_max_invoice' => 45,
  'piutang_max_invoice' => 55,
  'hutang_max_invoice' => 65,
  'pengeluaran_max_invoice' => 67,
]);
$websocket->SEND();
PHP Object Oriented Usage
$websocket = new WMI_websocket(false, false);
$websocket->createData([
