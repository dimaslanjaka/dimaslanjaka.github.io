Contoh Gulp NodeJS

gulp.task('default', function() {
  return gulp.src([
      'src/**/*',         // select all files
      '!src/**/_*/',      // exclude folders starting with '_'
      '!src/**/_*/**/*',  // exclude files/subfolders in folders starting with '_'
    ])
    .pipe(gulp.dest('dist'));
});
