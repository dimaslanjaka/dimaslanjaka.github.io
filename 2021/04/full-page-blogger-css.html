How to show only content on blogger page

<style>
  @media only screen and (min-width: 1025px) {

    #sidebar-wrapper,
    #midsidebar-wrapper,
    .gapad2,
    .blog-pager,
    .post-header-line-1,
    .post-footer {
      display: none !important;
    }

    #main-wrapper {
      width: 150% !important;
      /*98% default*/
    }

    .post {
      width: 150% !important;
      /*98% default*/
    }

    #comments {
      width: 150% !important;
      /*98% default*/
    }

    .post-title,
    .post-labels,
    post-icons,
    post-author {
      display: none !important;
    }
  }
</style>
